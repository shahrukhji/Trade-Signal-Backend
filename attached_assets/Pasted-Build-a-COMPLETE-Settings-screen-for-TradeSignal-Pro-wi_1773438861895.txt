    return {
      'Content-Type': 'application/json',
      'Accept': 'application/json',
      'X-UserType': 'USER',
      'X-SourceID': 'WEB',
      'X-ClientLocalIP': '127.0.0.1',

      'X-MACAddress': 'fe80::216e:6507:4b90:3719',
      'X-PrivateKey': this.apiKey,
    };
  }

  private buildAuthHeaders(): Record<string, string> {
    return {
      ...this.buildBaseHeaders(),
      'Authorization': `Bearer ${this.session?.jwtToken || ''}`,
    };

      if (step) Object.assign(step, update);
      onStepUpdate([...steps]);

      status: 'connecting',
      session: null,
      profile: null,
      funds: null,
      holdings: null,
      error: null,
      lastChecked: Date.now(),
      latencyMs: 0,
      tokenExpiresAt: 0,
      isDemo: false,
      validationSteps: steps,

    try {
      // ═══ STEP 1: Validate Format ═══
      updateStep('format', { status: 'running', message: 'Checking credential format...' });
      await delay(300);

      if (!credentials.clientCode || credentials.clientCode.trim().length < 2) {
        updateStep('format', { status: 'error', message: 'Client Code is required', detail: 'Enter your Angel One Client Code (e.g., S1234567 or R12345)' });
        state.status = 'error';
        state.error = 'Invalid Client Code';
        return state;
      }

      if (!credentials.pin || credentials.pin.trim().length < 4) {
        updateStep('format', { status: 'error', message: 'PIN must be at least 4 characters', detail: 'Enter your Angel One trading PIN (the same PIN you use to login to Angel One app)' });
        state.status = 'error';
        state.error = 'Invalid PIN';
        return state;
      }

      if (!credentials.apiKey || credentials.apiKey.trim().length < 6) {
        updateStep('format', { status: 'error', message: 'API Key must be at least 6 characters', detail: 'Get your API key from smartapi.angelone.in → My Apps → Create/View API Key. It is a long alphanumeric string (6-64 characters).' });
        state.status = 'error';
        state.error = 'Invalid API Key';
        return state;
      }

      if (!credentials.totp || credentials.totp.trim().length !== 6) {
        updateStep('format', { status: 'error', message: 'TOTP must be exactly 6 digits', detail: 'Open your authenticator app (Google Authenticator / similar) and enter the current 6-digit code for Angel One.' });
        state.status = 'error';
        state.error = 'Invalid TOTP';
        return state;

        updateStep('format', { status: 'error', message: 'TOTP must be 6 numeric digits only', detail: 'TOTP is the 6-digit number from your authenticator app. It changes every 30 seconds.' });
        state.status = 'error';
        state.error = 'TOTP must be numeric';
        return state;

        status: 'success',
        message: 'Credentials format valid ✓',

      updateStep('login', { status: 'running', message: 'Sending login request to Angel One...' });

      try {
        const loginResponse = await fetch(`${API_BASE}${ENDPOINTS.login}`, {
          method: 'POST',
          headers: this.buildBaseHeaders(),
          body: JSON.stringify({
            clientcode: credentials.clientCode.trim(),
            password: credentials.pin.trim(),
            totp: credentials.totp.trim(),
          }),
          signal: AbortSignal.timeout(20000),
        });

        const loginLatency = Date.now() - loginStartTime;
        state.latencyMs = loginLatency;

        const loginData = await loginResponse.json();

        if (loginResponse.ok && loginData.status === true && loginData.data) {
          this.session = {
            jwtToken: loginData.data.jwtToken,
            refreshToken: loginData.data.refreshToken,
            feedToken: loginData.data.feedToken,
          };
          state.session = this.session;
          state.tokenExpiresAt = Date.now() + (12 * 60 * 60 * 1000); // Session valid till midnight

          updateStep('login', {
            status: 'success',
            message: `Authenticated successfully ✓ (${loginLatency}ms)`,
            detail: `JWT Token: ${this.session.jwtToken.substring(0, 20)}...`
          });
        } else {
          // Handle specific error codes
          const errorCode = loginData.errorcode || '';
          const errorMsg = getErrorMessage(errorCode, loginData.message || 'Login failed');

          updateStep('login', {
            status: 'error',
            message: `Authentication failed: ${errorMsg}`,
            detail: `Error code: ${errorCode}. ${errorCode === 'AB1007' ? 'Double-check your Client Code and PIN.' : errorCode === 'AB1008' || errorCode === 'AB1003' ? 'Your TOTP may have expired. Open authenticator app and use the fresh current code.' : errorCode === 'AG8001' || errorCode === 'AB1010' ? 'Your API key may be invalid. Go to smartapi.angelone.in to verify.' : 'Check all your credentials and try again.'}`
          });

          state.status = 'error';
          state.error = errorMsg;
          return state;
        }
      } catch (networkError: any) {
        if (networkError.name === 'AbortError') {
          updateStep('login', { status: 'error', message: 'Login request timed out (20s)', detail: 'Angel One servers may be slow. Try again.' });
          state.status = 'error';
          state.error = 'Connection timeout';
          return state;
        }

        // Network error — offer demo mode
        console.warn('Angel One API unreachable:', networkError.message);
        this.isDemo = true;
        state.isDemo = true;

          status: 'warning',
          message: 'Cannot reach Angel One servers — Entering Demo Mode',
          detail: `Network error: ${networkError.message}. This may be due to CORS restrictions in browser/Electron. The app will use simulated data for demonstration.`

          refreshToken: 'demo_refresh_' + Date.now(),
          feedToken: 'demo_feed_' + Date.now(),

          brokerId: 'DEMO',

          status: 'warning',
          message: 'Using demo profile',

              status: 'success',
              message: `${state.holdings.holdings.length} holdings loaded ✓`,
              detail: `Total Value: ₹${state.holdings.totalHolding.totalHoldingValue.toLocaleString('en-IN')} | P&L: ₹${state.holdings.totalHolding.totalProfitAndLoss.toLocaleString('en-IN')} (${state.holdings.totalHolding.totalPnlPercentage.toFixed(2)}%)`
            });
          } else {
            const errCode = holdData.errorcode || '';
            updateStep('holdings', {
              status: 'warning',
              message: `Holdings: ${getErrorMessage(errCode, holdData.message || 'No holdings found')}`,
              detail: errCode === 'AB1004' ? 'Enable Portfolio permission in SmartAPI portal.' : 'You may not have any holdings yet.'
            });
            state.holdings = { holdings: [], totalHolding: { totalHoldingValue: 0, totalInvValue: 0, totalProfitAndLoss: 0, totalPnlPercentage: 0 } };
          }
        } catch (e: any) {
          updateStep('holdings', { status: 'warning', message: `Holdings error: ${e.message}` });
          state.holdings = { holdings: [], totalHolding: { totalHoldingValue: 0, totalInvValue: 0, totalProfitAndLoss: 0, totalPnlPercentage: 0 } };
        }
      }

      // ═══ FINAL STATUS ═══
      const hasErrors = steps.some(s => s.status === 'error');
      const loginSuccess = steps.find(s => s.id === 'login')?.status !== 'error';

      if (loginSuccess) {
        state.status = this.isDemo ? 'demo' : 'connected';
      } else {
        state.status = 'error';
      }

      state.validationSteps = steps;
      state.lastChecked = Date.now();

      // Start health monitoring
      if (state.status === 'connected' || state.status === 'demo') {
        this.startHealthMonitor();
      }

      return state;

    } catch (fatalError: any) {
      state.status = 'error';
      state.error = fatalError.message;
      return state;
    }
  }

  // ═══ REFRESH TOKEN ═══

  async refreshToken(): Promise<boolean> {
    if (this.isDemo || !this.session?.refreshToken) return this.isDemo;

    try {
      const resp = await fetch(`${API_BASE}${ENDPOINTS.generateToken}`, {
        method: 'POST',
        headers: this.buildAuthHeaders(),
        body: JSON.stringify({ refreshToken: this.session.refreshToken }),
        signal: AbortSignal.timeout(10000),
      });

      const data = await resp.json();
      if (resp.ok && data.status === true && data.data) {
        this.session = {
          jwtToken: data.data.jwtToken,
          refreshToken: data.data.refreshToken,
          feedToken: data.data.feedToken,
        };
        return true;
      }
      return false;
    } catch {
      return false;
    }
  }

  // ═══ LOGOUT ═══

  async logout(): Promise<void> {
    this.stopHealthMonitor();
    if (!this.isDemo && this.session?.jwtToken) {
      try {
        await fetch(`${API_BASE}${ENDPOINTS.logout}`, {
          method: 'POST',
          headers: this.buildAuthHeaders(),
          body: JSON.stringify({ clientcode: this.clientCode }),
        });
      } catch (_) {}
    }
    this.session = null;
    this.isDemo = false;
    this.apiKey = '';
    this.clientCode = '';
  }

  // ═══ HEALTH MONITOR ═══

  private startHealthMonitor(): void {
    this.stopHealthMonitor();
    this.healthInterval = setInterval(async () => {
      if (this.isDemo) {
        this.onStatusChange?.({ status: 'demo', lastChecked: Date.now(), latencyMs: Math.floor(Math.random() * 5 + 1) });
        return;
      }
      if (!this.session) return;

      const start = Date.now();
      try {
        const resp = await fetch(`${API_BASE}${ENDPOINTS.getProfile}`, {
          method: 'GET',
          headers: this.buildAuthHeaders(),
          signal: AbortSignal.timeout(10000),
        });
        const latency = Date.now() - start;

        if (resp.ok) {
          this.onStatusChange?.({ status: 'connected', lastChecked: Date.now(), latencyMs: latency });
        } else if (resp.status === 401) {
          const refreshed = await this.refreshToken();
          this.onStatusChange?.({ status: refreshed ? 'connected' : 'expired', lastChecked: Date.now(), latencyMs: latency });
        } else {
          this.onStatusChange?.({ status: 'error', lastChecked: Date.now(), latencyMs: latency });
        }
      } catch {
        this.onStatusChange?.({ status: 'error', lastChecked: Date.now(), latencyMs: Date.now() - start });
      }
    }, 30000);
  }

  private stopHealthMonitor(): void {
    if (this.healthInterval) { clearInterval(this.healthInterval); this.healthInterval = null; }
  }

  setOnStatusChange(callback: (state: Partial<ConnectionState>) => void): void {
    this.onStatusChange = callback;
  }
